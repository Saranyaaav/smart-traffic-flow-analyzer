Quick Start Integration

Kotlin Android Client 
// Initialize TrafficFlow API Client
val client = TrafficFlowClient.Builder()
    .baseUrl("https://api.trafficflow.ai/v1/")
    .apiKey("your_api_key_here")
    .enableWebSocket()
    .build()

// Get real-time traffic data
val trafficData = client.analytics.getCurrentTraffic()

// Subscribe to live updates
client.websocket.subscribe("traffic_update") { update ->
    updateUI(update.data)
}

// Detect vehicles in image
val detection = client.vehicles.detect(imageBytes)
println("Detected ${detection.vehicles.size} vehicles")

Build & Install

ü§ñ AI Service Setup

Python AI Service 
# Navigate to AI service directory

# Create virtual environment

source venv/bin/activate  # On Windows: venv\Scripts\activate

# Install dependencies
